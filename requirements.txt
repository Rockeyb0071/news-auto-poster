def build_caption(headline_titles: list) -> str:
    """
    headline_titles: list of English headline strings, order carousel
    slides ke order se match karta hai. Caption ke labels Hinglish
    mein hain, headlines English mein (original wording).
    """
    lines = ["Aaj ki Top Headlines\n"]
    for i, title in enumerate(headline_titles, start=1):
        lines.append(f"{i}. {title}")
    lines.append(f"\n☕ {PAGE_TAG}")
    lines.append("\n#news #india #dailynews #updates")
    return "\n".join(lines)
